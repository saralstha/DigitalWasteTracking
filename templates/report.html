async function submitReport() { const loc = document.getElementById('location').value; const type = document.getElementById('type').value; const weight = document.getElementById('weight').value; const file = document.getElementById('photo').files[0]; const formData = new FormData(); formData.append("location", loc); formData.append("type", type); formData.append("weight", weight); formData.append("photo", file); await fetch('/api/report', { method: 'POST', body: formData }); alert("Waste report submitted!"); document.getElementById("reportForm").reset(); loadReports(); }
